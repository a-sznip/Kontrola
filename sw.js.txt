self.addEventListener('install', event => {
  event.waitUntil(
    caches.open('kontrole-v1').then(cache => {
      // WAŻNE: Dodajemy wszystkie zasoby, z których korzysta strona, w tym biblioteki i ikony.
      return cache.addAll([
        '/',
        
        // Zmień poniższą ścieżkę na nazwę Twojego głównego pliku HTML:
        // Masz wgrane: index.txt, int4.html, int530.09.2025.html. Załóżmy, że to index.txt
        'index.txt', 
        
        '/manifest.json.txt', // Pamiętaj o rozszerzeniu .txt, jeśli tak wgrałeś
        '/sw.js.txt',         // Dodajemy samego Service Workera do bufora
        
        // Lokalnie wgrane ikony
        '/android-chrome-192x192.png',
        '/android-chrome-512x512.jpg',
        
        // Zewnętrzne biblioteki (CDN)
        'https://cdn.tailwindcss.com',
        'https://cdnjs.cloudflare.com/ajax/libs/jspdf/2.5.1/jspdf.umd.min.js',
        'https://cdnjs.cloudflare.com/ajax/libs/html2canvas/1.4.1/html2canvas.min.js',
        'https://fonts.googleapis.com/css2?family=Inter:wght@400;500;700&display=swap'
      ]);
    })
  );
});

self.addEventListener('fetch', event => {
  event.respondWith(
    caches.match(event.request).then(response => {
      return response || fetch(event.request);
    })
  );
});
