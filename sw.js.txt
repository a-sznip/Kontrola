self.addEventListener('install', event => {
  event.waitUntil(
    caches.open('kontrole-v1').then(cache => {
      return cache.addAll([
        '/',
        '/int4.html',
        '/manifest.json',
        // Dodaj tu ścieżki do innych plików, np. ikon, css, js
      ]);
    })
  );
});

self.addEventListener('fetch', event => {
  event.respondWith(
    caches.match(event.request).then(response => {
      return response || fetch(event.request);
    })
  );
});
